#include<stdio.h>

int factorial(int);

int main()
{
	int n,fact;
	
	printf("Enter the number you want to get factorial of:");
	scanf("%d",&n);
	
	fact=factorial(n);	
	
	printf("The factorial of %d is %d",n,fact);
}

int factorial(int a)
{
	int y=a-1;
	if(a==0)
	{
		return 1;
	}
	else
	{
		return (a*factorial(y));
	}
}
